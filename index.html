    // --- State Variables ---

    let currentQuestionIndex = 0;

    // --- Core Functions ---


    function initSpeechRecognition() {

      // Check for API support

      const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;

      

      if (!SpeechRecognition) {

        alert("Your browser does not support voice recognition. Please use Google Chrome or Edge.");

        return false;

      }


      // Check for secure context (Crucial for microphone access)

      if (location.protocol !== 'https:' && location.hostname !== 'localhost' && location.hostname !== '127.0.0.1') {

        httpsWarning.classList.remove('hidden');

      }


      recognition = new SpeechRecognition();

      recognition.lang = 'en-US'; // Set language to English

      recognition.continuous = false;

      recognition.interimResults = false;


      recognition.onresult = (event) => {

        const transcript = event.results[0][0].transcript;

        handleTranscription(transcript);

      };


      recognition.onerror = (event) => {

        console.error('Speech error:', event.error);

        stopRecording();

        let msg = "Error recording.";

        if(event.error === 'not-allowed') msg = "Microphone permission denied. Please allow mic access in your browser settings.";

        if(event.error === 'no-speech') msg = "No voice detected. Try again.";

        document.getElementById('recordStatus').textContent = msg;

      };


      recognition.onend = () => {

        stopRecording();

      };

      

      return true;

    }


    function startRecording() {

      if (!recognition) {

        alert("Speech engine not initialized. Please refresh and click 'Start Exercises' again.");

        return;

      }

      

      isRecording = true;

      recordBtn.classList.add('recording-active');

      document.getElementById('micIcon').classList.add('hidden');

      document.getElementById('waveAnimation').classList.remove('hidden');

      document.getElementById('recordStatus').textContent = "Listening...";

      

      // Crucial: Must be called in a user-gesture handler (click)

      try {

        recognition.start();

      } catch(e) {

        console.error("Error starting recognition", e);

        // If it fails, stop visual feedback

        stopRecording();

      }

    }


    function stopRecording() {

      isRecording = false;

      recordBtn.classList.remove('recording-active');

      document.getElementById('micIcon').classList.remove('hidden');

      document.getElementById('waveAnimation').classList.add('hidden');

      

      // Only reset text if we aren't showing results yet

      if (analysisCard.classList.contains('hidden')) {

        document.getElementById('recordStatus').textContent = "Tap to speak";

      }

    }


    function handleTranscription(transcript) {

      const currentQuestion = questions[currentQuestionIndex];

      const analysis = analyzeResponse(transcript, currentQuestion);

      

      // Display Transcription with color coding

      const transcribedDiv = document.getElementById('transcribedText');

      const words = transcript.split(/\s+/);

      

      let html = words.map(word => {

        const cleanWord = word.replace(/[.,!?;:'"]/g, '').toLowerCase();

        let statusClass = 'word-error'; // Default to error/wrong

        

        // Check if word matches expected answer

        if (currentQuestion.expectedAnswer.toLowerCase().includes(cleanWord)) {

          statusClass = 'word-correct';

        } 

        // Check if word is a keyword

        else if (currentQuestion.keywords.some(k => k.toLowerCase() === cleanWord)) {

          statusClass = 'word-correct';

        } 

        // Check if word is part of the question structure

        else {

          const expectedWords = currentQuestion.question.replace('___', currentQuestion.expectedAnswer).toLowerCase().split(/\s+/);

          if(expectedWords.some(ew => ew.includes(cleanWord))) {

             statusClass = 'word-warning'; 

          }

        }

        

        return `<span class="inline-block px-1 py-0.5 mx-0.5 rounded ${statusClass}">${word}</span>`;

      }).join(' ');

      

      transcribedDiv.innerHTML = html;


      // Display Scores

      document.getElementById('grammarScore').textContent = `${analysis.grammarScore}%`;

      document.getElementById('structureScore').textContent = `${analysis.structureScore}%`;

      document.getElementById('vocabularyScore').textContent = `${analysis.vocabularyScore}%`;


      // Display Detailed Feedback (Show correct answer HERE)

      const feedbackDiv = document.getElementById('feedbackContent');

      let scoreColor = analysis.overallScore >= 70 ? 'var(--success)' : analysis.overallScore >= 40 ? 'var(--warning)' : 'var(--error)';

      

      feedbackDiv.innerHTML = `

        <div class="p-4 rounded-lg" style="background-color: ${analysis.overallScore >= 70 ? 'rgba(102, 187, 106, 0.1)' : analysis.overallScore >= 40 ? 'rgba(255, 167, 38, 0.1)' : 'rgba(239, 83, 80, 0.1)'}">

          <p class="font-semibold mb-2" style="color: ${scoreColor}">

            ${analysis.overallScore >= 70 ? 'Excellent job!' : analysis.overallScore >= 40 ? 'Good effort' : 'Needs more practice'}

          </p>

          <p class="text-sm text-[var(--fg-muted)]">${currentQuestion.grammarFocus}</p>

        </div>

        

        <div class="bg-[var(--bg)] p-4 rounded-lg border border-[var(--border)]">

          <p class="text-xs text-[var(--fg-muted)] uppercase mb-1">Correct Answer:</p>

          <p class="text-lg font-semibold text-[var(--accent-green)]">${currentQuestion.expectedAnswer}</p>

          <p class="text-sm text-[var(--fg-muted)] mt-2 italic">"${currentQuestion.question.replace('___', currentQuestion.expectedAnswer)}"</p>

        </div>

      `;


      analysisCard.classList.remove('hidden');

      analysisCard.classList.add('slide-up');

      recordBtn.style.display = 'none'; // Hide record button after answering


      // Save results

      results.push({

        question: currentQuestion,

        transcript: transcript,

        analysis: analysis

      });


      totalScore += analysis.overallScore;

      document.getElementById('totalScore').textContent = totalScore;

    }


    function analyzeResponse(transcript, question) {

      const lowerTranscript = transcript.toLowerCase();

      const expected = question.expectedAnswer.toLowerCase();

      

      // 1. Grammar Score: Did they use the target structure?

      let grammarScore = 0;

      const hasTargetWord = lowerTranscript.includes(expected);

      const hasCorrectPattern = question.tensePattern.test(transcript);

      

      if (hasTargetWord && hasCorrectPattern) grammarScore = 100;

      else if (hasTargetWord || hasCorrectPattern) grammarScore = 60;

      else grammarScore = 20;


      // 2. Structure Score: Context words match

      const expectedWords = question.question.replace('___', question.expectedAnswer).toLowerCase().split(/\s+/);

      const transcriptWords = lowerTranscript.split(/\s+/);

      let matchCount = 0;

      

      transcriptWords.forEach(tw => {

        if(expectedWords.some(ew => tw.includes(ew) || ew.includes(tw))) matchCount++;

      });

      

      let structureScore = Math.round((matchCount / expectedWords.length) * 100);


      // 3. Vocabulary Score: Keywords used

      const keywordMatches = question.keywords.filter(kw => lowerTranscript.includes(kw.toLowerCase()));

      let vocabularyScore = Math.round((keywordMatches.length / question.keywords.length) * 100);


      return {

        grammarScore: Math.min(100, grammarScore),

        structureScore: Math.min(100, structureScore),

        vocabularyScore: Math.min(100, vocabularyScore),

        overallScore: Math.round((grammarScore * 0.5 + structureScore * 0.3 + vocabularyScore * 0.2))

      };

    }

    }

    }

      document.getElementById('finalScore').textContent = totalScore;

      

      const summaryList = document.getElementById('summaryList');

      summaryList.innerHTML = results.map((r, i) => {

        const s = r.analysis.overallScore;

        const color = s >= 70 ? 'var(--success)' : s >= 40 ? 'var(--warning)' : 'var(--error)';

        return `

          <div class="flex justify-between items-center p-3 bg-white rounded-lg border border-[var(--border)]">

            <span class="font-medium">Question ${i+1}</span>

            <span class="font-bold" style="color: ${color}">${s}%</span>

          </div>

        `;

      }).join('');

    }
